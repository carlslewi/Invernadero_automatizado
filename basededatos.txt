create database invernadero;

use invernadero;

create table sensores(id int unsigned not null primary key auto_increment,
nombre text not null);

create table valor_sensores(id int unsigned not null primary key autoincrement,
valor float not null,
fecha TIMESTAMP DEFAULT CURRENT_TIMESTAMP not null,
idsensor int unsigned not null,
foreign key (idsensor) references sensores(id) on delete cascade on update cascade);
